\section*{Research Synopsis} %%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
\titledate{\minititle{Nonprofit Governance: Gender Diversity on Boards}}{Nov 2018 - \present} \\
\subtext{With:} John Bush, Hannah Christine, Allison Russell, Femida Handy \\
\subtext{Summary:} Using data from IRS 990s, we determine whether the gender composition of nonprofit boards influences the adoption of good governance practices.

\insertspace

\titledate{\minititle{To Be or Not To Be Active: Game-Theoretic Decision-Making During Pandemics}}{Apr 2020 - \present} \\
\subtext{Advisors:} Aislinn Bohren, Ashwin Kambhampati \\
\subtext{Summary:} I model the trade-off between reducing the risk of spreading/catching the virus and being socially and economically active.

\insertspace 

\titledate{\minititle{Face to Face vs. Face to Screen: The Short-Term Evaluation of Telesecundarias}}{Apr 2020 - \present} \\
\subtext{Advisor:} Petra Todd \\
\subtext{Summary:} With a Generalized Roy Model, I quantify the effects of attending a telesecundaria relative to attending a traditional school in Mexico on ENLACE test scores in Spanish and Math.

\insertspace

\titledate{\minititle{Creating a Toilet Habit in Kenya}}{Jun 2019 - Jul 2019} \\
\subtext{With:} Ahmed Mushfiq Mobarak, Judy Chevalier, Johann Caro-Burnett \\
\subtext{Summary:} We implemented a large-scale field experiment in Nairobi, Kenya that observes toilet use before, during, and after various incentive schemes to understand how habits form and behavior persists. 

\insertspace

\titledate{\minititle{Intergenerational Transmission of Environmental Values}}{Mar 2018 - Jun 2019} \\
\subtext{With:} Femida Handy, Itay Greenspan, Tally Katz-Gerro \\
\subtext{Summary:} We explored parental environmental behaviors and socialization domains as predictors for child environmental behavior by using national samples of parent-child dyads in Israel, South Korea, and the United States.

\insertspace

\titledate{\minititle{The Influence of Strategic Potential of a Tennis Game on Effort}}{Oct 2018 - Dec 2019} \\ 
\subtext{Advisor:} Francis DiTraglia \\
\subtext{Summary:} Using point-by-point data and player statistics from the ATP website, I studied the likelihood of winning a tennis match when losing a game intentionally would be strategically beneficial.

\insertspace

\titledate{\minititle{Ephemerality of Preferences}}{Sep 2017 - Dec 2017} \\
\subtext{With:} Justin Mignatti ; \subtext{Advisor:} Irina Pimenova \\
\subtext{Summary:} We determined whether preferences change within two weeks by comparing the results of baseline and endline surveys that ask participants to rank a various sets of goods.
